O ENDEREÇO DO BANCO (PODE FICAR VISÍVEL) // ===================================================================== const LINK_DO_MOTOR = "https://motor-osint-privado.analise-seguranca.workers.dev"; // ===================================================================== // ⚙️ 2. FUNÇÃO PARA BUSCAR OS DOSSIÊS (COM BLINDAGEM OPSEC) // ===================================================================== async function carregarDossies() { // 🛑 A JOGADA DE MESTRE: O sistema abre uma caixa perguntando a senha SÓ PARA VOCÊ. // Ela nunca fica salva no GitHub! const senhaDigitada = prompt("🚨 ACESSO RESTRITO (NÍVEL COMANDANTE):\nDigite a Chave Mestra do Bunker:"); // Se você cancelar ou não digitar nada, ele trava. if (!senhaDigitada) { alert("🚨 Acesso Cancelado."); return; } // Montando a chave blindada na hora const CHAVE_BUNKER = "Bearer " + senhaDigitada; try { // O Bunker bate na porta do Cloudflare com a senha que você acabou de digitar const resposta = await fetch(LINK_DO_MOTOR, { method: "GET", headers: { "Authorization": CHAVE_BUNKER } }); // Se o hacker digitar qualquer lixo na caixa, a Muralha rejeita. if (resposta.status === 401) { alert("🚨 ALERTA DE INTRUSÃO: Senha Mestra Incorreta. Acesso Negado."); return; } // Os dados chegando em segurança const dadosBunker = await resposta.json(); // ===================================================================== // 📊 3. INJETANDO OS DADOS NA SUA TABELA PRETA // ===================================================================== const corpoTabela = document.getElementById("corpo-tabela"); if (!corpoTabela) { console.log("Aviso: Tabela não encontrada na tela."); return; } corpoTabela.innerHTML = ""; // Desenhando os relatórios dadosBunker.forEach(dossie => { let dados = typeof dossie === 'string' ? JSON.parse(dossie) : dossie; let linha = ` ${dados.protocolo} ${dados.perfil} ${dados.canal} ${dados.mensagem} ${dados.anexo} ${dados.data} `; corpoTabela.innerHTML += linha; }); } catch (erro) { console.error("🚨 Falha na comunicação com a base:", erro); alert("Erro de conexão com o satélite Cloudflare."); } } // ===================================================================== // 🚀 ACIONA A TRANCA ASSIM QUE A PÁGINA CARREGA // ===================================================================== window.onload = carregarDossies;
